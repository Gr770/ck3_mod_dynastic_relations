dr_debug_interaction_print_dynastic_relation = {
	category = interaction_category_friendly
	common_interaction = yes

	auto_accept = yes
	
	is_shown = {
        NOT = { scope:recipient = scope:actor }
    }
	
	is_valid = {
        dr_can_dynastic_relation_be_established_betwen_characters = {
            CHARACTER_A = scope:actor
            CHARACTER_B = scope:recipient
        }
	}

	
	on_accept = {

        dr_save_dynastic_relation_between_dynasties_in_scope = {
            DYNASTY_A = scope:actor.dynasty
            DYNASTY_B = scope:recipient.dynasty
            SCOPE_NAME = dynastic_relation_val
        } 


        scope:actor = {
            
            save_scope_value_as = {
                name = hello
                value = 10.0
            }

            debug_log_scopes = yes
            debug_log = "Hello = [Character.MakeScope.GetValue('hello')]"
        }
        
        
        #debug_log = "Dynastic relation with this dynasty = [SCOPE.GetValue('hello')]"
        
	}
}
